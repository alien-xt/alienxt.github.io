RDD Operations

Transformation

// rdd create by array
val num = sc.parallelize(1 to 100)
// map: return new array
num.map(_ + 1).collect
// filter: even numbers
num.filter(i => i % 2 == 0).collect
// rdd create by array
val num1 = sc.parallelize(List(List(1,2),List(3,4)))
// flatMap: foreach every element，return new array
num1.flatMap(_.map(_+1)).collect
// sample: true，是否放回  0.3，抽样比例 random，随机种子
import java.util.Random
num.sample(true, 0.3,new Random().nextInt())

// kv 
val kv1 = sc.parallelize(List(("A",1),("B",2),("C",3),("A",4),("B",5)))
// sortByKey
kv1.sortByKey().collect
// groupByKey
kv1.groupByKey().collect
// reduceByKey
kv1.reduceByKey(_+_).collect

Spark SQL

import org.apache.spark.sql.SQLContext
import org.apache.spark.sql.hive.HiveContext

Json

RDD to DataFrame

// build context
val sqlContext = new SQLContext(sc)
// create rdd from json str
val rdd = sc.parallelize(
      """[{"name":"alien","age":"26","address":{"city":"Xiamen","state":"China"}},{"name":"guoxiatong","age":"18","address":{"city":"Beijing","state":"China"}}]""" :: Nil)

// create dataframe from rdd
val df = sqlContext.read.json(rdd)
// registe table "people"
df.registerTempTable("people")

// show content of the df
df.show()
// show schema in a tree format
df.printSchema()
// show the "name" cloumn
df.select("name").show()
// show muti cloumn, and operation
df.select(df("name"), df("age") + 1).show()
// select people older than 21
df.filter(df("age") > 21).show()
// group by
df.groupBy("age").count().show()
// save result: if in cluster model, save on the hdfs
df.select("name").write.format("json").save("output/spark/people_result_1")

// who is teenager
val teenagers = sqlContext.sql("SELECT name, age FROM people WHERE age >= 13 AND age <= 21")
// print query result
teenagers.collect().foreach(println)
// save result
teenagers.write.format("json").save("output/spark/people_result_2")
